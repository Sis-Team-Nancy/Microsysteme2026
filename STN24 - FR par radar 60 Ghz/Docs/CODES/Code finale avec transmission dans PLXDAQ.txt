#include "DFRobot_HumanDetection.h"
#include <SPI.h>
#include <RadioLib.h>

// ================= RADAR =================
DFRobot_HumanDetection hu(&Serial1);

// ================= LORA SX1262 =================
// ---- Pins XIAO ESP32S3 ----
#define LORA_CS    41
#define LORA_DIO1  39
#define LORA_RESET 42
#define LORA_BUSY  40

SX1262 radio = new Module(LORA_CS, LORA_DIO1, LORA_RESET, LORA_BUSY);

void setup() {

  Serial.begin(9600);
  Serial1.begin(115200, SERIAL_8N1, 44, 43);

  delay(2000);

  // ===== INITIALISATION PLXDAQ (EXCEL) =====
  Serial.println("CLEARDATA");
  Serial.println("LABEL,Temps,Respiration (rpm),Frequence cardiaque (bpm)");

  // -------- Initialisation Radar --------
  Serial.println("Début initialisation radar");

  while (hu.begin() != 0) {
    Serial.println("Erreur d'initialisation !!!");
    delay(1000);
  }

  Serial.println("Radar initialisé");

  while (hu.configWorkMode(hu.eSleepMode) != 0) {
    Serial.println("Erreur mode !!!");
    delay(1000);
  }

  hu.configLEDLight(hu.eHPLed, 1);
  hu.sensorRet();

  Serial.println("Radar prêt");

  // -------- Initialisation LoRa --------
  Serial.println("Initialisation SX1262...");

  int state = radio.begin(868.0, 125.0, 7, 5, 0x12, 14, 8);

  if (state == RADIOLIB_ERR_NONE) {
    Serial.println("LoRa prêt !");
  } 
  else {
    Serial.print("Erreur LoRa : ");
    Serial.println(state);
    while (true);
  }

  Serial.println();
}

void loop() {

  // -------- Lecture Radar --------
  int respiration = hu.getBreatheValue();
  int heart = hu.getHeartRate();

  Serial.print("Respiration rate: ");
  Serial.print(respiration);
  Serial.println(" rpm");

  Serial.print("Heart rate: ");
  Serial.print(heart);
  Serial.println(" bpm");

  // ===== ENVOI DES DONNEES VERS EXCEL (PLX-DAQ) =====
  Serial.print("DATA,TIME,");
  Serial.print(respiration);
  Serial.print(",");
  Serial.println(heart);

  // -------- Construction message LoRa --------
  String trame = "R:" + String(respiration) +
                 ",F:" + String(heart) + "#";

  Serial.print("Envoi LoRa : ");
  Serial.println(trame);

  // -------- Transmission LoRa --------
  int state = radio.transmit(trame);

  if (state == RADIOLIB_ERR_NONE) {
    Serial.println("Transmission OK");
  } 
  else {
    Serial.print("Erreur TX: ");
    Serial.println(state);
  }

  Serial.println();
  delay(1000); // mesure toutes les 1s
}
